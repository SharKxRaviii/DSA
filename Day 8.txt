title: Longest Subarray with given Sum K(Positives)

Problem Statement: Given an array nums of size n and an integer k, find the length of the longest sub-array that sums to k. If no such sub-array exists, return 0.

Examples
Example 1:
Input:
 nums = [10, 5, 2, 7, 1, 9], k = 15  
Output:
 4  
Explanation:
 The longest sub-array with a sum equal to 15 is [5, 2, 7, 1], which has a length of 4. This sub-array starts at index 1 and ends at index 4, and the sum of its elements (5 + 2 + 7 + 1) equals 15. Therefore, the length of this sub-array is 4.

Example 2:
Input:
 nums = [-3, 2, 1], k = 6  
Output:
 0  
Explanation:
 There is no sub-array in the array that sums to 6. Therefore, the output is 0.

solution:
let arr = [10, 5, 2, 7, 1, 9];
let k = 15;

function longestSubArr (nums, k) {
  let maxLen=0;
  
  for(let i=0; i<nums.length; i++) {
    let sum = 0;
    
    for(let j=i; j<nums.length; j++) {
      sum += nums[j];
      
      if(sum === k) {
        let length = j - i + 1;
        maxLen= Math.max(maxLen, length);
      }
    }
  }
  
  return maxLen;
}

console.log(longestSubArr(arr, k));
// output: 4
